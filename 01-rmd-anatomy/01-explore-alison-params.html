# By site

```{r include=FALSE}
(by_site <- mockdata %>% 
  group_by(site) %>% 
  summarize(
    total_site = sum(n),
    total_target_site = sum(n_target),
    prop_enroll = (total_site / total_target_site)
    )
 )
```

```{r}
ggplot(by_site, 
       aes(
         x = site, 
         y = prop_enroll, 
         fill = site)) +
  geom_col(show.legend = FALSE) +
  scale_y_continuous(name = "Enrollment percentage",
                     labels = scales::percent_format(accuracy = 1)) +
  expand_limits(y = 1) +
  coord_flip()
```

# By site and arm

```{r include=FALSE}
(by_site_arm <- mockdata %>% 
  group_by(site, arm) %>% 
  summarize(
    total_site_arm = sum(n),
    total_target_site_arm = sum(n_target),
    prop_enroll = (total_site_arm / total_target_site_arm)
    )
 )
```

```{r}
ggplot(by_site_arm,
       aes(x = site, 
           y = prop_enroll, 
           group = interaction(arm, site),
           fill = arm)) +
  geom_col(position = position_dodge()) +
  scale_y_continuous(name = "Enrollment percentage",
                     labels = scales::percent_format(accuracy = 1)) +
  expand_limits(y = 1)
```
